# SECCIÓN 1 — ÍNDICE COMPUESTO (bit_calculation)   [status: verified_walk]
==============================================================================
# El dato de mayor valor. El bitcalc_validation del propio manifest TKC ya
# probó que el bit-map de catálogo (shelf 31-28 / slot 27-24) es INCORRECTO
# (40/40 disagree). La fórmula de abajo REPRODUCE el walk real y debe REEMPLAZAR
# ese bit-map como decoder de índice.
#
# Hay DOS espacios de índice, uno por base OID:

index_space: ONU-ID
  applies_to_base: .1.3.6.1.4.1.3902.1082
  formula: suffix = 285278208 + slot*256 + pon        # 0x11010000 + slot*0x100 + pon
  onu_id: se anexa como octeto OID final (2 niveles): <prefix>.<suffix>.<onuID>
  shelf: fijo = 1 (codificado en el prefijo 0x11)
  ranges: slot 1..30, pon 1..16, onu 1..N

index_space: TYPE
  applies_to_base: .1.3.6.1.4.1.3902.1012
  formula: suffix = 268435456 + slot*65536 + pon*256  # 0x10000000 + slot*0x10000 + pon*0x100
  onu_id: se anexa como octeto OID final: <prefix>.<suffix>.<onuID>
  shelf: fijo = 1 (codificado en el prefijo 0x10)

# EVIDENCIA DE VALIDACIÓN (cruza con docs/walks/ZTE_C300_ifnames.txt):
#   ifIndex 285278721 = ifName "gpon_1/2/1"  → slot=2, pon=1
#   check: 285278208 + 2*256 + 1 = 285278721  ✓  (coincide con el walk)
#   ifIndex 285278722 = "gpon_1/2/2"          → 285278208 + 2*256 + 2 = 285278722 ✓
# Ejemplos de sufijo (pon=1): slot1=285278465 | slot2=285278721 | slot3=285278977
# TYPE (pon=1):               slot1=268501248 | slot2=268566784 | slot3=268632320
